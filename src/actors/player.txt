actor cursor_base : Weapon
{
  States
  {
    Spawn:
      TNT1 A -1
      Stop    
    Select:
      TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_Raise
      Loop
    Deselect:
      TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
      Loop
    Ready:
      TNT1 A 1 A_WeaponReady
      Loop
    Fire:
      TNT1 A 1 ACS_NamedExecuteAlways("Select_Puff", 0)
      TNT1 A 1 A_ReFire
      TNT1 A 1
      Goto Ready
    AltHold:
    Hold:
      TNT1 A 1
      TNT1 A 1 A_ReFire
      TNT1 A 1
      Goto Ready
    AltFire:
      TNT1 A 1 A_SelectWeapon("Cursor_Point")
      TNT1 A 1 A_ReFire
      Goto Ready
  }
}

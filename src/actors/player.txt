actor cursor_base : Weapon
{
  RenderStyle "None"
  
  States
  {
    Spawn:
      CLIP A -1
      Stop    
    Select:
      CLIP AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_Raise
      Loop
    Deselect:
      CLIP AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
      Loop
    Ready:
      CLIP A 1 A_WeaponReady
      Loop
    Fire:
      CLIP A 1 
      CLIP A 1 A_ReFire
      CLIP A 1
      Goto Ready
    AltHold:
      CLIP A 1
      CLIP A 1 A_ReFire
      CLIP A 1
      Goto Ready
    Hold:
      CLIP A 1
      CLIP A 1 A_ReFire
      CLIP A 1
      Goto Ready
    AltFire:
      CLIP A 1 A_SelectWeapon("Cursor_Point")
      CLIP A 1 A_ReFire
      Goto Ready
  }
}
